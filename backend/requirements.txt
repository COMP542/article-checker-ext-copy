# Install everything with:
#   pip install -r requirements.txt
#
# After installing, also run this once to download the spaCy language model:

flask          # the web server framework
flask-cors     # allows the browser extension to make requests to Flask
python-dotenv  # reads the .env file so we don't hardcode the API key

requests       # makes HTTP requests to NewsAPI

sentence-transformers  # loads the AI embedding model (all-MiniLM-L6-v2)
                       # converts article text into vectors for comparison
scikit-learn           # provides cosine_similarity for scoring
numpy                  # math on vectors and arrays
matplotlib             # heat map / similarity visualization

textblob       # tone analysis — scores subjectivity and polarity of writing
spacy          # grammar parser — used to detect passive voice constructions
